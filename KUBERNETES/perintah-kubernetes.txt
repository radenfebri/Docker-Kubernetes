----------------------------- Install kubernetes Dashboard------------------------------
https://github.com/kubernetes/dashboard (cara install kubernetes dashboard)
kubectl apply -f https://raw.githubusercontent.com/kubernetes/dashboard/v2.6.0/aio/deploy/recommended.yaml
kubectl proxy --port=80 (terserah portnya)
http://localhost:8001/api/v1/namespaces/kubernetes-dashboard/services/https:kubernetes-dashboard:/proxy/#/login
kubectl apply -f .\service-account.yaml (jalankan ini)
kubectl apply -f .\ClusterRoleBinding.yaml (jalankan ini)
kubectl -n kubernetes-dashboard create token admin-user
kubectl -n kubernetes-dashboard delete serviceaccount admin-user (delete user)
kubectl -n kubernetes-dashboard delete clusterrolebinding admin-user (delete cluster)





---------------------------- Perintah kubernetes ---------------------------------------
kubectl create deployment nginx-depl --image=nginx (membuat deployment nginx)
kubectl get deployment (menampilkan hasil deployment)
kubectl get pod (menampilkan pod)
kubectl get pods -o wide (menampilkan lebih detail tentang pods)
kubectl logs namapod (melihat logs nginx)
kubectl logs namapod (melihat logs mongo)
kubectl describe pod mongo-depl-85dcbc595b-wz45j (untuk mengetahui pod)
kubectl delete -n default deployment mongo-depl (untuk delete deployment "mongo-depl = nama deployment")
kubectl delete -n default pod mongo-depl-85dcbc595b-wz45j (untuk delete pod "mongo-depl-85dcbc595b-wz45j = nama pod")
kubectl exec -it mongo-depl-85dcbc595b-wz45j -- bin/bash (mengakses mongo melalui exec)
kubectl create -f filepod.yaml (cara bikin pod melalui yaml)
kubectl port-forward namapod portAkses:portPod
kubectl port-forward nginx 8080:80 (contoh akes pod testing)
kubectl get pods --show-labels (melihat pods + labels)
kubectl label pod nginx environment=development (menambahan label pada pods yang belum mempunyai label)
kubectl label pod nginx environment=update --overwrite (untuk mengupdate label)
kubectl get pods -l environment (mencari label yang ada environment)
kubectl get pods -l '!environment' (mencari label yang tidak punya label environment)
kubectl get pods -l 'environment in (production, development)' (mencari berdasrkan key yang detail)
